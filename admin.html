// Supabase에서 관리자 비밀번호 가져오기 async function loadAdminPassword() { try { const res = await fetch(`${SUPABASE_URL}/rest/v1/admin_settings?key=eq.admin_password&select=value`, { headers: { 'apikey': SUPABASE_KEY, 'Authorization': `Bearer ${SUPABASE_KEY}` } }); const data = await res.json(); if (data && data.length > 0) { ADMIN_PASSWORD = data[0].value; } } catch (e) { console.error('관리자 비밀번호 로드 실패', e); } } // Supabase에 관리자 비밀번호 저장 async function saveAdminPassword(newPw) { await fetch(`${SUPABASE_URL}/rest/v1/admin_settings?key=eq.admin_password`, { method: 'PATCH', headers: { 'apikey': SUPABASE_KEY, 'Authorization': `Bearer ${SUPABASE_KEY}`, 'Content-Type': 'application/json', 'Prefer': 'return=representation' }, body: JSON.stringify({ value: newPw }) }); }
